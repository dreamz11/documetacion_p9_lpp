def porcentaje_hidratos()
  porcentaje_hidratos = 0.0

  @conjunto_alimentos.each do |i|
    porcentaje_hidratos += i.carbohidratos
  end

  ((porcentaje_hidratos / 100.ceil(2)) * 100).ceil(2)
end
