def calculo_hombre(alimento)
  energia = 0.0
  kgCO2eq = 0.0
  m2terreno = 0.0
  gramos = 0.0

  gramos = (100.to_f * 54.to_f) / alimento.proteinas.to_f
  energia = (alimento.energia.to_f * gramos.to_f) / 100.to_f
  kgCO2eq = (alimento.kgCO2eq.to_f * gramos.to_f) / 1000.to_f
  m2terreno = (alimento.m2terreno.to_f * gramos.to_f) / 1000.to_f

  "(#{alimento.nombre},#{gramos} gramos consumidos ,#{energia} kcal, impacto de kgC02 #{kgCO2eq}, impacto de m2terreno #{m2terreno} )"
end

def calculo_varios_alimentos(alimento1, alimento2, alimento3, genero)
  energia1 = 0.0
  energia2 = 0.0
  energia3 = 0.0
  kgCO2eq1 = 0.0
  kgCO2eq2 = 0.0
  kgCO2eq3 = 0.0
  m2terreno1 = 0.0
  m2terreno2 = 0.0
  m2terreno3 = 0.0
  gramos1 = 0.0
  gramos2 = 0.0
  gramos3 = 0.0

  if genero == "mujer"
    gramos1 = (100.to_f * 10.to_f) / alimento1.proteinas.to_f
    gramos2 = (100.to_f * 10.to_f) / alimento2.proteinas.to_f
    gramos3 = (100.to_f * 21.to_f) / alimento3.proteinas.to_f
  else
    gramos1 = (100.to_f * 12.to_f) / alimento1.proteinas.to_f
    gramos2 = (100.to_f * 12.to_f) / alimento2.proteinas.to_f
    gramos3 = (100.to_f * 30.to_f) / alimento3.proteinas.to_f
  end

  energia1 = (alimento1.energia.to_f * gramos1.to_f) / 100.to_f
  energia2 = (alimento2.energia.to_f * gramos2.to_f) / 100.to_f
  energia3 = (alimento3.energia.to_f * gramos3.to_f) / 100.to_f
  kgCO2eq1 = (alimento1.kgCO2eq.to_f * gramos1.to_f) / 1000.to_f
  kgCO2eq2 = (alimento2.kgCO2eq.to_f * gramos2.to_f) / 1000.to_f
  kgCO2eq3 = (alimento3.kgCO2eq.to_f * gramos3.to_f) / 1000.to_f
  m2terreno1 = (alimento1.m2terreno.to_f * gramos1.to_f) / 1000.to_f
  m2terreno2 = (alimento2.m2terreno.to_f * gramos2.to_f) / 1000.to_f
  m2terreno3 = (alimento3.m2terreno.to_f * gramos3.to_f) / 1000.to_f

  "( [#{alimento1.nombre},#{alimento2.nombre},#{alimento3.nombre} ], [#{gramos1} , #{gramos2} , #{gramos3} ] = #{gramos1 + gramos2 + gramos3} gramos consumidos,[#{energia1.ceil(2)}, #{energia2.ceil(2)}, #{energia3.ceil(2)} ] = #{(energia1 + energia2 + energia3)} kcal totales ,  [#{kgCO2eq1.ceil(2)}, #{kgCO2eq2.ceil(2)},#{kgCO2eq3.ceil(2)}] = #{(kgCO2eq1 + kgCO2eq2 + kgCO2eq3).ceil(2)} impacto de kgC02  ,  [ #{m2terreno1} ,#{m2terreno2} ,#{m2terreno3} ] = #{m2terreno1 + m2terreno2 + m2terreno3} impacto de m2terreno )"
end
