def huella_nutricional
  result_energia = []
  result_carbono = []
  indice_energia = 0
  indice_carbono = 0

  each() do |n|
    if n.energia < 100
      result_energia << 1
    elsif n.energia >= 760 && n.energia <= 830
      result_energia << 2
    else result_energia << 3       end

    if n.kgCO2eq < 100
      result_carbono << 1
    elsif n.kgCO2eq >= 800 && n.kgCO2eq <= 1200
      result_carbono << 2
    else result_carbono << 3       end
  end

  result_energia.each do |n|
    indice_energia += n
  end
  result_carbono.each do |n|
    indice_carbono += n
  end

  indice_carbono = indice_carbono / result_carbono.length
  indice_energia = indice_energia / result_energia.length

  (indice_energia + indice_carbono) / 2
end

def uso_terreno()
  terreno = 0.0

    terreno += i.m2terreno
