def insert(data)
  node = Node.new data
  unless head
    self.head = node
  else
    node.prev = self.tail
    self.tail.next = node
  end
  self.tail = node
  @length += 1
end

def remove_head()
  node_aux = @head
  @head = node_aux.next
  @head.prev = nil
  node_aux.next = nil

  @length -= 1

  return node_aux
end

def to_s
